# Qwen3-4B-Instruct GRPO runs — synsql, gen-1 dialects, DDP-3.
#
# Both runs use the (a+b+c) hyperparam set (--num-generations 6, --lr-scheduler-type
# cosine, --max-steps 500) on the multi-GPU pipeline (3-rank colocate vLLM, no
# Unsloth). Bypass run_grpo_sql_ddp.sh because in trl-0.22 colocate mode the
# launcher's reserved "vLLM GPU" is unused -- driving accelerate launch directly
# with --num_processes=3 utilises all 3 GPUs.
#
# Inherited defaults (set in train/grpo_sql_distributed.py / TrainArgs):
#   model_name      : Qwen/Qwen3-4B-Instruct-2507  (upstream Qwen, not Unsloth)
#   lora_rank       : 32
#   load_in_4bit    : false   (DDP path runs bf16, no 4-bit)
#   learning_rate   : 5e-6
#   grad_accum      : 1
#   coverage_mode   : partition
#   temperature     : 1.6     (now configurable; default matches grpo_sql.py)
#   synsql_size     : 1000    (~440 tasks survive the complexity filter; 243
#                              kept after the length filter)
#   synsql_complexity: simple,moderate
#   save_steps      : 250     (overridden below to ease disk pressure)
#
# trl-0.22 invariant for DDP: per_device_bs * num_processes * grad_accum must
# divide num_generations. With 3 ranks and num_gens=6 we use per_device_bs=2 ->
# generation_batch=6 = 1 prompt x 6 gens, matching the single-GPU runs.
#
# Required additions to train/grpo_sql_distributed.py for this session:
#   - --generator synsql      (was wikisql/bird/golden/eval_suite only)
#   - --lr-scheduler-type     (was hardcoded "linear")
#   - --temperature           (was hardcoded 1.0; default now 1.6)
#
# Launch from /workspace/manysql. HF_HOME=/tmp/hf-home so the model is read
# from the existing cache (the /workspace cache hits the same per-pod quota
# the training outputs do); absolute --output-dir paths land outside /tmp.

##############################################################################
# 1.  Synsql, 1-dialect (snowflake_clone), DDP-3 on GPUs 3,4,5.
##############################################################################

CUDA_VISIBLE_DEVICES=3,4,5 \
HF_HOME=/tmp/hf-home HF_DATASETS_CACHE=/tmp/hf-datasets \
nohup /tmp/venv-gpu/bin/python -m accelerate.commands.launch \
    --num_processes 3 --main_process_port 29501 \
    -m train.grpo_sql_distributed \
    --generator synsql \
    --dialects snowflake_clone \
    --max-steps 500 --num-generations 6 --per-device-train-batch-size 2 \
    --lr-scheduler-type cosine --save-steps 250 \
    --max-seq-length 6144 --no-wandb \
    --output-dir /workspace/manysql/outputs/grpo_qwen3_4b_sql_synsql_snowflake_orig_ddp3 \
    > /tmp/grpo_synsql_qwen3_snowflake_orig_ddp3.log 2>&1 &

##############################################################################
# 2.  Synsql, 10-dialect (gen-1 set: snowflake_clone + 9 most-recent dialects
#     by `git log --diff-filter=A` order at the time of the abc runs),
#     DDP-3 on GPUs 0,1,2.
##############################################################################

CUDA_VISIBLE_DEVICES=0,1,2 \
HF_HOME=/tmp/hf-home HF_DATASETS_CACHE=/tmp/hf-datasets \
nohup /tmp/venv-gpu/bin/python -m accelerate.commands.launch \
    --num_processes 3 --main_process_port 29500 \
    -m train.grpo_sql_distributed \
    --generator synsql \
    --dialects snowflake_clone,sqlite_clone,bigmaria_pivot,bqserver_qualify_filter,mariflake_comma_semi,mydb2_rollup,mysqlite_upsert,orashift_merge,pgserver_schema_ns,redgres_lateral \
    --max-steps 500 --num-generations 6 --per-device-train-batch-size 2 \
    --lr-scheduler-type cosine --save-steps 250 \
    --max-seq-length 6144 --no-wandb \
    --output-dir /workspace/manysql/outputs/grpo_qwen3_4b_sql_synsql_10dialects_orig_ddp3 \
    > /tmp/grpo_synsql_qwen3_10dialects_orig_ddp3.log 2>&1 &
